import os from flask import Flask, jsonify, request app = Flask(__name__) @app.route('/') def home(): return jsonify({"message": "Сервер працює!"}) # Приклад hello @app.route('/hello/') def hello(name): return jsonify({"message": f"Привіт, {name}!"}) # POST з JSON @app.route('/data', methods=['POST']) def data(): content = request.get_json() if not content: return jsonify({"error": "Будь ласка, надішліть JSON"}), 400 return jsonify({"received": content}) # Запуск сервера if __name__ == '__main__': port = int(os.environ.get("PORT", 5000)) # Render задає PORT app.run(host="0.0.0.0", port=port)
